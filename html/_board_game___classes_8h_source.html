249template <typename T>
250class GameManager {
251 Board<T>* boardPtr;
252 Player<T>* players[2];
253 UI<T>* ui;
254
255public:
259 GameManager(Board<T>* b, Player<T>* p[2], UI<T>* u)
260 : boardPtr(b), ui(u) {
261 players[0] = p[0];
262 players[1] = p[1];
263 players[0]->set_board_ptr(b);
264 players[1]->set_board_ptr(b);
265 }
266
270 void run() {
271 ui->display_board_matrix(boardPtr->get_board_matrix());
272 Player<T>* currentPlayer = players[0];
273
274 while (true) {
275 for (int i : {0, 1}) {
276 currentPlayer = players[i];
277 Move<T>* move = ui->get_move(currentPlayer);
278
279 while (!boardPtr->update_board(move))
280 move = ui->get_move(currentPlayer);
281
282 ui->display_board_matrix(boardPtr->get_board_matrix());
283
284 if (boardPtr->is_win(currentPlayer)) {
285 ui->display_message(currentPlayer->get_name() + " wins!");
286 return;
287 }
288 if (boardPtr->is_lose(currentPlayer)) {
289 ui->display_message(players[1 - i]->get_name() + " wins!");
290 return;
291 }
292 if (boardPtr->is_draw(currentPlayer)) {
293 ui->display_message("Draw!");
294 return;
295 }
296 }
297 }
298 }
